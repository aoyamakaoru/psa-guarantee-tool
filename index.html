最低保証買取ツール（PSA9が何枚までなら利益が出る？ + PSA10率入力）

期待値として「10枚数/9枚数」を表示します（端数あり）
0にすると「損益分岐でOK」
「最低でも利益○円欲しい」→ここに入れる
「とにかく赤字回避」→0
「計算」を押すと結果が出ます。
const yen = (x)=> (isFinite(x) ? Math.round(x).toLocaleString('ja-JP') : '—'); const num = (x)=> (isFinite(x) ? (Math.round(x*100)/100).toLocaleString('ja-JP') : '—'); function clamp(x, min, max){ return Math.max(min, Math.min(max, x)); } function clampInt(x, min, max){ x = Math.floor(x); if (x < min) return min; if (x > max) return max; return x; } function calc(){ const c = Number(document.getElementById('card').value); const g = Number(document.getElementById('grading').value); const n = Math.max(1, Math.floor(Number(document.getElementById('n').value))); const r10 = clamp(Number(document.getElementById('rate10').value), 0, 100); const p10 = Number(document.getElementById('p10').value); const p9 = Number(document.getElementById('p9').value); const target = Number(document.getElementById('target').value); const cost = c + g; const diff = p10 - p9; // 全部10だった場合の利益 const profitAll10 = n * p10 - n * cost; // 9の最大許容枚数（整数） let yMax; if (diff <= 0){ yMax = n; } else { yMax = Math.floor((profitAll10 - target) / diff); yMax = clampInt(yMax, 0, n); } const min10 = n - yMax; // 利益達成に必要な最低10枚数（整数） const needRate10 = (min10 / n) * 100; // 必要10率（%） // yMax枚9、残り10のときの利益 const profitAtYMax = (n - yMax) * p10 + yMax * p9 - n * cost; // PSA10率入力から期待値（端数あり） const exp10 = n * (r10 / 100); const exp9 = n - exp10; // 期待値での利益（端数あり） const expProfit = exp10 * p10 + exp9 * p9 - n * cost; // 期待値ベースで赤字回避できそうか（目標利益も含める） const expOk = expProfit >= target; const out = document.getElementById('out'); out.innerHTML = `
PSA9は ${yMax}枚 までなら目標利益OK 必要PSA10枚数 ${min10}枚 / 必要PSA10率 ${num(needRate10)}%
（枚数n=${n}のとき。「9が${yMax}枚」=「10が${min10}枚」）
PSA10率 ${num(r10)}% の期待値 → PSA10: ${num(exp10)}枚 / PSA9: ${num(exp9)}枚
期待値での利益: ${yen(expProfit)}円（目標: ${yen(target)}円）
1枚あたり原価: ${yen(cost)}円 / 10-9差: ${yen(diff)}円
全部10の利益: ${yen(profitAll10)}円 / 全部9の利益: ${yen(n * p9 - n * cost)}円
「9が${yMax}枚」のときの利益: ${yen(profitAtYMax)}円
${diff > 0 ? `
※ 9が1枚増えるごとに利益が ${yen(diff)}円 減ります
` : `
※ PSA9が同額以上なので、9が増えても利益は減りません
` }
※「期待値」は端数を含む概算です（実際は整数の枚数になります）
`; } document.getElementById('calc').addEventListener('click', calc); document.getElementById('fill').addEventListener('click', ()=>{ document.getElementById('card').value = 3000; document.getElementById('grading').value = 2500; document.getElementById('n').value = 10; document.getElementById('rate10').value = 60; document.getElementById('p10').value = 12000; document.getElementById('p9').value = 6000; document.getElementById('target').value = 0; calc(); }); // 入力したら即再計算（好みで外してOK） ["card","grading","n","rate10","p10","p9","target"].forEach(id=>{ document.getElementById(id).addEventListener('input', calc); }); calc();
